50  EllipsoidParameter(double ia, double ib)
51  {
52  if (ib > 1000000) // ib 是短半轴
53  {
54  a = ia;
55  b = ib;
56 
57  f = (a - b) / a;
58  e2 = (a * a - b * b) / (a * a);
59  ep2 = (a * a - b * b) / (b * b);
60  }
61  else if (ib < 1) // ib 是椭球第一偏心率的平方
62  {
63  a = ia;
64  e2 = ib;

67  ep2 = (a * a - b * b) / (b * b);

69  }
70 
71  c = a / (1 - f);

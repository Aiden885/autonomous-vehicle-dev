1.首先安装protobuf3.6.1

tar -xvf protobuf-all-3.6.1.tar.gz
cd protobuf-3.6.1
./autogen.sh               #生成configure文件
./configure --prefix=/usr  #设置安装路径为/usr
make -j 8                  #编译并设置线程数
make check                 #可以省略，不执行
sudo make install          #安装
protoc --version           #检查版本


2.安装c语言的protoc-c，版本适配protoc3.6.1，最后安装的是1.3.3版本

tar -zxvf protobuf-c-1.3.3.tar.gz
cd protobuf-c-1.3.3
./configure
make
sudo make install
sudo ldconfig  # 刷新系统库缓存
